统计出现次数最多的字符


        var str = 'abcoocfjhqkoosjfococasjop' ;
        var o = {};
        for(var i = 0;i < str.length ; i++) {
            var chars =str.charAt(i);
            if(o[chars]) {
                o[chars]++;
            } else {
                o[chars] = 1
            }
        }
        var max = 0;
        var ch = '';
        for (var k in o) {
            if (o[k] > max) {
                max = o[k];
                ch = k
            }
        }
        console.log("最多的是" + ch +"出现" + max + "次");
