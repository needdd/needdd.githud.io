                var text = document.querySelector('input');
                text.onfocus = function() {
                    if (this.value === '手机'){
                    this.value = '';
                    this.style.color = 'red';
                    }
                }
                text.onblur = function() {
                    if (this.value === '') {
                        this.value = '手机';
                        this.style.color = '#999'
                    }
                }

节点操作

node.parentNode 返回最近的父节点
parentNode.childNodes 子节点（获取全部节点 元素节点 文本节点）不常用
parentNode.children 只读属性节点
parentNode.firstChild 获取第一个子节点
parentNode.lastChild 获取最后一个子节点
parentNode.firstElementChild 获取第一个元素节点 (IE9 UP)
parentNode.lastElementChild 获取最后一个元素节点 (IE9 UP)
parentNode.chilren[0] 获取第一个元素节点 (开发常用)
parentNode.chilren[parentNode.length - 1] 获取最后一个元素节点 (开发常用)

node.nextSibling 获取下一个兄弟节点 （包含文本节点）
node.previousSibling 获取上一个兄弟节点 （包含文本节点）
node.nextElementSibling 获取下一个元素兄弟节点 (IE9 UP)
node.previousElementSibling 获取上一个元素兄弟节点 (IE9 UP)

<script>
var but = document.querySelector('button');
var text = document.querySelector('textarea');
var ul = document.querySelector('ul')
but.onclick = function() {
    if(text.value == '') {
        alert('输入内容');
    } else {
        var li = document.createElement('li');
        li.innerHTML = text.value + "<a href='javascript:;'>删除</a>";
        // ul.appendChild(li);
        ul.insertBefore(li,ul.children[0]);
        text.value = '';
        var as = document.querySelectorAll('a');
        for(var i = 0; i < as.length; i++) {
            as[i].onclick = function () {
                ul.removeChild(this.parentNode);
            }
        }
    }
}
</script>
